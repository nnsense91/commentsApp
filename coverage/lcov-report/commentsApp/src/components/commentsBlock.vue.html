			:comments="comments"
		)

<script>
import { mapState } from 'vuex';
 
export default {
	components: {
		commentsList: () => import('./commentsList'),
		commentsAddNew: () => import('./commentsAddNew'),
	},
	data() {
		return {
			isLegendShow: false,
		};
	},
	computed: {
		/**
			* Возвращает данные из store
		*/
		...mapState('comments', {
			comments: (state) => state.comments,
		}),
	},
	methods: {
		showLegendOn() {
			this.isLegendShow = true;
		},
		showLegendOff() {
			this.isLegendShow = false;
		},
	},
};
